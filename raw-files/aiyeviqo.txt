-- Funciones de animación
local function animateButton(button, active)
    local tweenInfo = TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
    
    if active then
        local tween = TweenService:Create(button, tweenInfo, {
            BackgroundColor3 = COLOR_ACCENT,
            TextColor3 = COLOR_PRIMARY
        })
        tween:Play()
    else
        local tween = TweenService:Create(button, tweenInfo, {
            BackgroundColor3 = COLOR_SECONDARY,
            TextColor3 = COLOR_TEXT
        })
        tween:Play()
    end
end

-- SISTEMA DE ARRASTRE MEJORADO PARA MÓVIL Y PC
local dragging = false
local dragStartPos
local frameStartPos

-- Función para manejar inicio de arrastre (táctil y mouse)
local function startDragging(input)
    if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStartPos = input.Position
        frameStartPos = MainFrame.Position
        return true
    end
    return false
end

-- Función para manejar movimiento durante arrastre
local function updateDragging(input)
    if dragging then
        local delta = input.Position - dragStartPos
        MainFrame.Position = UDim2.new(
            frameStartPos.X.Scale,
            frameStartPos.X.Offset + delta.X,
            frameStartPos.Y.Scale,
            frameStartPos.Y.Offset + delta.Y
        )
    end
end

-- Función para detener arrastre
local function stopDragging()
    dragging = false
end

-- Conectar eventos para PC (mouse)
TitleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        startDragging(input)
    end
end)

TitleBar.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        updateDragging(input)
    end
end)

-- Conectar eventos para móvil (táctil)

-- Efecto hover en botones (solo para PC)
local function setupHoverEffect(button)
    local originalColor = button.BackgroundColor3
    
    button.MouseEnter:Connect(function()
        if not (button == FreezeTradeButton and freezeActive) and not (button == ForceAcceptButton and forceActive) then
            local tween = TweenService:Create(button, TweenInfo.new(0.2), {
                BackgroundColor3 = Color3.fromRGB(
                    math.floor(originalColor.R * 255 * 1.2),
                    math.floor(originalColor.G * 255 * 1.2),
                    math.floor(originalColor.B * 255 * 1.2)
                )
            })
            tween:Play()
        end
    end)
    
    button.MouseLeave:Connect(function()
        if not (button == FreezeTradeButton and freezeActive) and not (button == ForceAcceptButton and forceActive) then
            local tween = TweenService:Create(button, TweenInfo.new(0.2), {
                BackgroundColor3 = originalColor
            })
            tween:Play()
        end
    end)
end

setupHoverEffect(FreezeTradeButton)
setupHoverEffect(ForceAcceptButton)
setupHoverEffect(CloseButton)
