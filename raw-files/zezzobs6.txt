-- SISTEMA DE ARRASTRE FUNCIONAL (GUI INDIVIDUAL)
local dragging = false
local dragInput, dragStart, startPos

-- Función para actualizar posición durante arrastre
local function update(input)
    local delta = input.Position - dragStart
    MainFrame.Position = UDim2.new(
        startPos.X.Scale, 
        startPos.X.Offset + delta.X,
        startPos.Y.Scale, 
        startPos.Y.Offset + delta.Y
    )
end

-- Conectar eventos de input para arrastre (ÁREA DEL TÍTULO)
local function connectDragEvents(frame)
    frame.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = MainFrame.Position
            
            -- Efecto visual al agarrar
            local GrabTween = TweenService:Create(
                MainFrame,
                TweenInfo.new(0.2),
                {BackgroundTransparency = 0.05}
            )
            GrabTween:Play()
            
            -- Capturar el fin del arrastre
            local connection
            connection = input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                    connection:Disconnect()
                    
                    -- Efecto visual al soltar
                    local ReleaseTween = TweenService:Create(
                        MainFrame,
                        TweenInfo.new(0.2),
                        {BackgroundTransparency = 0.1}
                    )
                    ReleaseTween:Play()
                end
            end)
        end
    end)
    
    frame.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)
end

-- Conectar eventos de arrastre al área del título y al marco de arrastre
connectDragEvents(DragArea)
connectDragEvents(TitleLabel)

-- Sistema de seguimiento de arrastre (NO AFECTA EL JUEGO)
UserInputService.InputChanged:Connect(function(input)
    if dragging and (input == dragInput) then
        -- Solo mover la GUI, no afectar el juego
        update(input)
    end
end)

-- Función para cambiar estado visual (sin funcionalidad real)
local function toggleIndicator(indicatorFrame, statusLabel, buttonName)
    if indicatorFrame.BackgroundColor3 == Color3.fromRGB(255, 50, 50) then
        -- Cambiar a activado (verde brillante)
        indicatorFrame.BackgroundColor3 = Color3.fromRGB(50, 255, 100)
        statusLabel.Text = "ACTIVE"
        statusLabel.TextColor3 = Color3.fromRGB(50, 255, 100)
        
        -- Efecto de pulso al activar
        local PulseTween = TweenService:Create(
            indicatorFrame,
            TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.Out, 0, true),
            {Size = UDim2.new(0, 22, 0, 22)}
        )
        PulseTween:Play()
    else
        -- Cambiar a desactivado (rojo)
        indicatorFrame.BackgroundColor3 = Color3.fromRGB(255, 50, 50)
        statusLabel.Text = "OFFLINE"
        statusLabel.TextColor3 = Color3.fromRGB(255, 80, 80)
    end
end
