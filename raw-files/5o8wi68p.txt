local function MakeDraggable(frame, handle)
    local dragging, dragInput, dragStart, startPos
    handle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = frame.Position
        end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
            local delta = input.Position - dragStart
            frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
        end
    end)
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = false
        end
    end)
end

-- [[ CORE LOGIC ]] --

-- This part runs every frame inside the RunService.Stepped connection
RunService.Stepped:Connect(function()
    if lp.Character and lp.Character:FindFirstChild("HumanoidRootPart") then
        local hrp = lp.Character.HumanoidRootPart
        
        -- Fake Lag Logic:
        -- If enabled, it forces the HumanoidRootPart to stay anchored (frozen in space)
        if OnyxConfig.FakeLag then
            hrp.Anchored = true
        else
            -- Ensure it unanchors when the toggle is turned off
            hrp.Anchored = false
        end
    end
    
    -- Anti-Ragdoll logic kept in stepped to ensure it doesn't kill you
    if lp.Character and lp.Character:FindFirstChild("Humanoid") then
        local hum = lp.Character.Humanoid
        if OnyxConfig.AntiRagdoll then
            hum.PlatformStand = false
            hum:SetStateEnabled(Enum.HumanoidStateType.Ragdoll, false)
            hum:SetStateEnabled(Enum.HumanoidStateType.FallingDown, false)
            if hum:GetState() == Enum.HumanoidStateType.Ragdoll then
                hum:ChangeState(Enum.HumanoidStateType.GettingUp)
            end
        end
    end
end)

-- Render Loop for Physics/Visuals
RunService.RenderStepped:Connect(function()
    if not lp.Character or not lp.Character:FindFirstChild("HumanoidRootPart") then return end
    local hrp = lp.Character.HumanoidRootPart
    
    -- REAL Fling on Touch Fix: Heavy Anchor Bypass
    if OnyxConfig.FlingOnTouch then
        local targetNear = false
        for _, p in pairs(Players:GetPlayers()) do
            if p ~= lp and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
                if (hrp.Position - p.Character.HumanoidRootPart.Position).Magnitude < 4.5 then
                    targetNear = true
                    break
                end
            end
        end

        if targetNear then
            -- Apply infinite mass to obliterate anyone who touches you
            hrp.CustomPhysicalProperties = PhysicalProperties.new(math.huge, 0.3, 0.5, 1, 1)
            -- Spin fast enough to break their physics
            hrp.AssemblyAngularVelocity = Vector3.new(0, 75000, 0)
            -- Force your character downwards so the collision impact doesn't send YOU to the moon
            hrp.AssemblyLinearVelocity = Vector3.new(0, -100, 0)
        else
            -- Instantly reset so you can walk completely normally when nobody is touching you
            hrp.CustomPhysicalProperties = PhysicalProperties.new(0.7, 0.3, 0.5, 1, 1)
        end
    else
        -- Ensure properties reset if toggled off mid-fling
        if hrp.CustomPhysicalProperties and hrp.CustomPhysicalProperties.Density == math.huge then
            hrp.CustomPhysicalProperties = PhysicalProperties.new(0.7, 0.3, 0.5, 1, 1)
        end
    end

    -- Invisible Mode
    if OnyxConfig.InvisibleMode then
        for _, part in pairs(lp.Character:GetDescendants()) do
            if part:IsA("BasePart") or part:IsA("Decal") then
                if part.Name ~= "HumanoidRootPart" then part.Transparency = 1 end
            end
        end
    end

    -- Anti Negative
    if OnyxConfig.AntiNegative then
        Lighting.Brightness = 2
        for _, v in pairs(Lighting:GetChildren()) do
            if v:IsA("ColorCorrectionEffect") or v:IsA("BlurEffect") then v.Enabled = false end
        end
        lp.PlayerGui:SetAttribute("InvertedControls", false)
        camera.FieldOfView = 70
    end
end)

end)
