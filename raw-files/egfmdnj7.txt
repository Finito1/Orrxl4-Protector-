-- VARIABLES
local flying = false
local flyBV, flyBG
local godMode = false

-- FUNCTIONS

local function getChar()
	return player.Character or player.CharacterAdded:Wait()
end

local function fly()
	if flying then return end
	flying = true
	
	local char = getChar()
	local hrp = char:WaitForChild("HumanoidRootPart")
	local hum = char:WaitForChild("Humanoid")
	
	flyBV = Instance.new("BodyVelocity", hrp)
	flyBV.MaxForce = Vector3.new(9e9,9e9,9e9)
	flyBV.Velocity = Vector3.new(0,0,0)
	
	flyBG = Instance.new("BodyGyro", hrp)
	flyBG.MaxTorque = Vector3.new(9e9,9e9,9e9)
	flyBG.CFrame = hrp.CFrame
	
	hum.PlatformStand = true
	
	UIS.InputBegan:Connect(function(i,g)
		if g then return end
		if i.KeyCode == Enum.KeyCode.W then
			flyBV.Velocity = hrp.CFrame.LookVector * 80
		end
	end)
end

local function unfly()
	flying = false
	local char = getChar()
	local hum = char:FindFirstChildOfClass("Humanoid")
	if flyBV then flyBV:Destroy() end
	if flyBG then flyBG:Destroy() end
	if hum then hum.PlatformStand = false end
end

local function god()
	local hum = getChar():WaitForChild("Humanoid")
	hum.MaxHealth = 10000
	hum.Health = 10000
	godMode = true
end

local function ungod()
	local hum = getChar():WaitForChild("Humanoid")
	hum.MaxHealth = 100
	hum.Health = 100
	godMode = false
end

local function unlockPrompt()
	for _,v in pairs(workspace:GetDescendants()) do
		if v:IsA("ProximityPrompt") then
			v.HoldDuration = 0
		end
	end
end

-- COMMAND EXECUTION
startBtn.MouseButton1Click:Connect(function()
	local text = box.Text
	
	if text == "/fly" then
		fly()
	elseif text == "/tptobase" then
		local char = getChar()
		char:MoveTo(player.RespawnLocation and player.RespawnLocation.Position or Vector3.new(0,5,0))
	elseif text == "/tpfuse" then
		local fuse = workspace:FindFirstChild("FuseMachine", true)
		if fuse then
			getChar():MoveTo(fuse:GetModelCFrame().Position)
		end
	elseif string.find(text, "/speed") then
		local num = tonumber(string.split(text," ")[2])
		if num and num >=1 and num <=500 then
			getChar():WaitForChild("Humanoid").WalkSpeed = num
		end
	elseif string.find(text, "/jump") then
		local num = tonumber(string.split(text," ")[2])
		if num and num >=1 and num <=500 then
			getChar():WaitForChild("Humanoid").JumpPower = num
		end
	elseif text == "/god" then
		god()
	elseif text == "/unlockprompt" then
		unlockPrompt()
	end
end)

stopBtn.MouseButton1Click:Connect(function()
	unfly()
	ungod()
	getChar():WaitForChild("Humanoid").WalkSpeed = 16
	getChar():WaitForChild("Humanoid").JumpPower = 50
end)

cat.MouseButton1Click:Connect(function()
	listFrame.Visible = not listFrame.Visible
end)

close.MouseButton1Click:Connect(function()
	gui:Destroy()
end)

local minimized = false
minimize.MouseButton1Click:Connect(function()
	main.Visible = false
	minimized = true
end)

UIS.InputBegan:Connect(function(input,gp)
	if gp then return end
	if input.KeyCode == Enum.KeyCode.RightShift and minimized then
		main.Visible = true
		minimized = false
	end
end)
