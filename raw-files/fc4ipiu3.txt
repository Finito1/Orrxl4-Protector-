local BlockedStates = {
    [Enum.HumanoidStateType.Ragdoll] = true,
    [Enum.HumanoidStateType.FallingDown] = true,
    [Enum.HumanoidStateType.Physics] = true,
    [Enum.HumanoidStateType.Dead] = true
}

local RemoteKeywords = { "useitem", "combatservice", "ragdoll" }
local Frozen = false
local DisabledRemotes = {}
local RemoteWatcher

local function ForceNormal(character)
    local hum = character:FindFirstChildOfClass("Humanoid")
    local hrp = character:FindFirstChild("HumanoidRootPart")
    if not hum or not hrp then return end
    hum.Health = hum.MaxHealth
    hum:ChangeState(Enum.HumanoidStateType.RunningNoPhysics)
    if not Frozen then
        Frozen = true
        hrp.Anchored = true
        hrp.AssemblyLinearVelocity = Vector3.zero
        hrp.AssemblyAngularVelocity = Vector3.zero
        hrp.CFrame += Vector3.new(0, 1.5, 0)
    end
end

local function Release(character)
    local hrp = character:FindFirstChild("HumanoidRootPart")
    if hrp and Frozen then
        hrp.Anchored = false
        Frozen = false
    end
end

local function RestoreMotors(character)
    for _, v in character:GetDescendants() do
        if v:IsA("Motor6D") then v.Enabled = true end
        if v:IsA("Constraint") then v.Enabled = false end
    end
end

local function InitAntiRagdoll(character)
    local hum = character:WaitForChild("Humanoid", 10)
    if not hum then return end
    
    for state in pairs(BlockedStates) do
        hum:SetStateEnabled(state, false)
    end
    
    hum.StateChanged:Connect(function(_, new)
        if BlockedStates[new] then
            ForceNormal(character)
            RestoreMotors(character)
        end
    end)
    
    RunService.Stepped:Connect(function()
        if BlockedStates[hum:GetState()] then
            ForceNormal(character)
        else
            Release(character)
        end
        hum.Health = hum.MaxHealth
    end)
end

local function KillRemote(remote)
    if not getconnections or not remote:IsA("RemoteEvent") then return end
    if DisabledRemotes[remote] then return end
    local name = remote.Name:lower()
    for _, key in RemoteKeywords do
        if name:find(key) then
            DisabledRemotes[remote] = {}
            for _, c in getconnections(remote.OnClientEvent) do
                if c.Disable then
                    c:Disable()
                    table.insert(DisabledRemotes[remote], c)
                end
            end
            break
        end
    end
end

local function InitAntiItem()
    local PlayerModule = require(player:WaitForChild("PlayerScripts"):WaitForChild("PlayerModule"))
    local Controls = PlayerModule:GetControls()
    Controls:Enable()
    
    for _, obj in ReplicatedStorage:GetDescendants() do KillRemote(obj) end
    RemoteWatcher = ReplicatedStorage.DescendantAdded:Connect(KillRemote)
end

-- ESP completo (Players + Secret/OG + Base Timer/Highlight gris)
local espBrainrotConnections = {}
local Animals = nil
local rarePets = {}
local playerPlot = nil

local function initializeBrainrotESP()
    local success, result = pcall(function()
        return require(ReplicatedStorage:WaitForChild("Datas").Animals)
    end)
    
    if success then
        Animals = result
        rarePets = {}
        
        for petName, petData in Animals do
            if petData.Rarity == "Secret" or petData.Rarity == "OG" then
                table.insert(rarePets, petName)
            end
        end
    end
end

local function findPlayerBase()
    local plots = workspace:FindFirstChild("Plots")
    if not plots then return nil end
    for _, plot in plots:GetChildren() do
        local sign = plot:FindFirstChild("PlotSign")
        if sign and sign:FindFirstChild("YourBase") and sign.YourBase.Enabled then
            return plot
        end
    end
    return nil
end

local function isPlayerBase(model)
    if not playerPlot then return false end
    local cur = model
    while cur and cur ~= workspace do
        if cur == playerPlot then return true end
        cur = cur.Parent
    end
    return false
end

local function createPetTag(model, petName, rarity)
    local ex = model:FindFirstChild("PetNameTag")
    if ex then ex:Destroy() end
    
    local bb = Instance.new("BillboardGui")
    bb.Name = "PetNameTag"
    bb.Size = UDim2.new(0,200,0,50)
    bb.StudsOffset = Vector3.new(0,5,0)
    bb.AlwaysOnTop = true
    bb.Parent = model
    
    local txt = Instance.new("TextLabel", bb)
    txt.Size = UDim2.new(1,0,1,0)
    txt.BackgroundTransparency = 1
    txt.Text = petName
    txt.TextColor3 = Color3.new(1,1,1)
    txt.TextStrokeTransparency = 0
    txt.TextStrokeColor3 = Color3.new(0,0,0)
    txt.TextScaled = true
    txt.Font = Enum.Font.Code
    
    if rarity == "Secret" then txt.TextColor3 = Color3.new(0.8, 0, 1)
    elseif rarity == "OG" then txt.TextColor3 = Color3.new(1, 0, 0) end
end

local function checkRarePet(model)
    if not Animals or isPlayerBase(model) then return end
    local n = model.Name
    if n:find("67") then return end
    
    for _, pet in rarePets do
        if n:find(pet) or n == pet then
            local d = Animals[pet]
            if d then createPetTag(model, pet, d.Rarity) break end
        end
    end
end

local function scanPlotsForPets()
    local plots = workspace:FindFirstChild("Plots")
    if not plots then return end
    local function rec(p)
        for _, c in p:GetChildren() do
            if c:IsA("Model") then checkRarePet(c) end
            if c:IsA("Folder") or c:IsA("Model") then rec(c) end
        end
    end
    rec(plots)
end

local function removePetTags()
    local plots = workspace:FindFirstChild("Plots")
    if not plots then return end
    local function rec(p)
        for _, c in p:GetChildren() do
            local t = c:FindFirstChild("PetNameTag")
            if t then t:Destroy() end
            if c:IsA("Folder") or c:IsA("Model") then rec(c) end
        end
    end
    rec(plots)
end

local function startPetESP()
    if not Animals then
        initializeBrainrotESP()
        task.wait(2)
    end
    if #rarePets == 0 then return end
    
    playerPlot = findPlayerBase()
    scanPlotsForPets()
    
    if espBrainrotConnections.desc then espBrainrotConnections.desc:Disconnect() end
    espBrainrotConnections.desc = workspace.DescendantAdded:Connect(function(d)
        if not Settings.ESP then return end
        if d:IsA("Model") then
            task.wait(0.2)
            if d.Parent and d.Parent.Parent and d.Parent.Parent.Name == "Plots" then
                checkRarePet(d)
            end
        end
    end)
end

local function stopPetESP()
    for _, c in espBrainrotConnections do if c then c:Disconnect() end end
    espBrainrotConnections = {}
    removePetTags()
end

local function updateBaseESP()
    local plots = workspace:FindFirstChild("Plots")
    if not plots then return end
    
    for _, plot in plots:GetChildren() do
        local purchases = plot:FindFirstChild("Purchases")
        local plotBlock = purchases and purchases:FindFirstChild("PlotBlock")
        local main = plotBlock and plotBlock:FindFirstChild("Main")
        
        local timeLabel = main and main:FindFirstChild("BillboardGui") and main.BillboardGui:FindFirstChild("RemainingTime")
        
        if timeLabel and main then
            local bb = baseEspInstances[plot.Name] or createBaseESP(plot, main)
            bb:FindFirstChildWhichIsA("TextLabel").Text = timeLabel.Text
        elseif baseEspInstances[plot.Name] then
            baseEspInstances[plot.Name]:Destroy()
            local hl = plot:FindFirstChildWhichIsA("Highlight")
            if hl then hl:Destroy() end
            baseEspInstances[plot.Name] = nil
        end
    end
end

local function removePlayerESP()
    for _,p in Players:GetPlayers() do
        if p.Character and p.Character:FindFirstChild("Head") then
            local e = p.Character.Head:FindFirstChild("PlayerESP")
            if e then e:Destroy() end
        end
    end
end

local function toggleESP()
    Settings.ESP = not Settings.ESP
    
    if Settings.ESP then
        for _,p in Players:GetPlayers() do addPlayerESP(p) end
        Players.PlayerAdded:Connect(function(p)
            p.CharacterAdded:Connect(function() task.wait(0.3) addPlayerESP(p) end)
        end)
        
        startPetESP()
        
        if not espBaseThread then
            espBaseThread = RunService.RenderStepped:Connect(updateBaseESP)
        end
    else
        removePlayerESP()
        stopPetESP()
        
        if espBaseThread then
            espBaseThread:Disconnect()
            espBaseThread = nil
        end
        for name, inst in baseEspInstances do
            inst:Destroy()
            local plot = workspace.Plots:FindFirstChild(name)
            if plot then
                local hl = plot:FindFirstChildWhichIsA("Highlight")
                if hl then hl:Destroy() end
            end
        end
        baseEspInstances = {}
    end
end

-- Infinite Jump
local function updateInfiniteJump()
    if Settings.InfiniteJump then
        if not fallClampConnection then
            fallClampConnection = RunService.Heartbeat:Connect(function()
                local char = player.Character
                if not char then return end
                local hrp = char:FindFirstChild("HumanoidRootPart")
                if hrp and hrp.Velocity.Y < -Settings.clampFallSpeed then
                    hrp.Velocity = Vector3.new(hrp.Velocity.X, -Settings.clampFallSpeed, hrp.Velocity.Z)
                end
            end)
        end
        
        if not infinityJumpConnection then
            infinityJumpConnection = UserInputService.JumpRequest:Connect(function()
                local char = player.Character
                if not char then return end
                local hrp = char:FindFirstChild("HumanoidRootPart")
                if hrp then
                    hrp.Velocity = Vector3.new(hrp.Velocity.X, Settings.jumpForce, hrp.Velocity.Z)
                end
            end)
        end
    else
        if infinityJumpConnection then infinityJumpConnection:Disconnect() infinityJumpConnection = nil end
        if fallClampConnection then fallClampConnection:Disconnect() fallClampConnection = nil end
    end
end

-- Booster (29 speed / 60 jump)
local function toggleBooster()
    Settings.Booster = not Settings.Booster
    
    local function apply(char)
        local hum = char:FindFirstChildOfClass("Humanoid")
        if hum then
            hum.WalkSpeed = Settings.Booster and Settings.boostSpeed or Settings.normalSpeed
            hum.JumpPower = Settings.Booster and Settings.boostJump or Settings.normalJump
            hum.UseJumpPower = true
        end
    end
    
    if player.Character then apply(player.Character) end
    
    player.CharacterAdded:Connect(function(char)
        task.wait(0.1)
        apply(char)
    end)
end

RunService.Heartbeat:Connect(function()
    if not Settings.Booster then return end
    local char = player.Character
    if not char then return end
    local hum = char:FindFirstChildOfClass("Humanoid")
    local root = char:FindFirstChild("HumanoidRootPart")
    if not hum or not root then return end
    
    if hum.MoveDirection.Magnitude > 0 then
        root.Velocity = Vector3.new(
            hum.MoveDirection.X * Settings.boostSpeed,
            root.Velocity.Y,
            hum.MoveDirection.Z * Settings.boostSpeed
        )
    end
end)

-- AUTO GRAB (sin barra visible)
task.spawn(function()
    local hrp, humanoid
    
    local function updateChar()
        local char = player.Character or player.CharacterAdded:Wait()
        hrp = char:WaitForChild("HumanoidRootPart", 8)
        humanoid = char:WaitForChild("Humanoid", 8)
    end
    
    updateChar()
    player.CharacterAdded:Connect(updateChar)
    
    local function getPos(prompt)
        local p = prompt.Parent
        if p:IsA("BasePart") then return p.Position end
        if p:IsA("Model") then
            local prim = p.PrimaryPart or p:FindFirstChildWhichIsA("BasePart")
            return prim and prim.Position
        end
        if p:IsA("Attachment") then return p.WorldPosition end
        return nil
    end
    
    local function buscarPromptCerca()
        local cerca, distMin = nil, math.huge
        local plots = workspace:FindFirstChild("Plots")
        if not plots or not hrp then return nil, math.huge end
        
        for _, obj in plots:GetDescendants() do
            if obj:IsA("ProximityPrompt") and obj.Enabled and obj.ActionText == "Steal" then
                local pos = getPos(obj)
                if pos then
                    local dist = (hrp.Position - pos).Magnitude
                    if dist <= obj.MaxActivationDistance + 6 then
                        if dist < distMin then
                            distMin = dist
                            cerca = obj
                        end
                    end
                end
            end
        end
        return cerca, distMin
    end
    
    local function activar(prompt)
        if prompt then
            fireproximityprompt(prompt, 20, math.huge)
            prompt:InputHoldBegin()
            task.wait(0.010)
            prompt:InputHoldEnd()
        end
    end
    
    local promptActual = nil
    local ultimoRobo = 0
    
    RunService.Heartbeat:Connect(function()
        if not Settings.Grab then return end
        if tick() - ultimoRobo > 0.06 or not promptActual then
            promptActual = buscarPromptCerca()
        end
    end)
    
    while true do
        if Settings.Grab and humanoid and hrp then
            if promptActual then
                local _, dist = buscarPromptCerca()
                if dist and dist <= promptActual.MaxActivationDistance + 6 then
                    activar(promptActual)
                    ultimoRobo = tick()
                    promptActual = nil
                    task.wait(0.020)
                end
            end
        end
        task.wait(0.1)
    end
end)
