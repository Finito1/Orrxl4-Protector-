-- Função para criar botão toggle
local function createToggle(text, positionY)
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(0.9,0,0,40)
    frame.Position = UDim2.new(0.05,0,0,positionY)
    frame.BackgroundColor3 = Color3.fromRGB(40,40,60)
    frame.Parent = main
    
    local corner = Instance.new("UICorner", frame)
    corner.CornerRadius = UDim.new(0,8)

    local label = Instance.new("TextLabel")
    label.Size = UDim2.new(0.7,0,1,0)
    label.BackgroundTransparency = 1
    label.Text = text
    label.TextColor3 = Color3.new(1,1,1)
    label.Font = Enum.Font.Gotham
    label.TextSize = 14
    label.Parent = frame

    local toggle = Instance.new("TextButton")
    toggle.Size = UDim2.new(0,40,0,20)
    toggle.Position = UDim2.new(0.8,0,0.5,-10)
    toggle.BackgroundColor3 = Color3.fromRGB(80,80,100)
    toggle.Text = ""
    toggle.Parent = frame
    
    local tcorner = Instance.new("UICorner", toggle)
    tcorner.CornerRadius = UDim.new(1,0)

    local enabled = false
    toggle.MouseButton1Click:Connect(function()
        enabled = not enabled
        toggle.BackgroundColor3 = enabled and Color3.fromRGB(0,170,255) or Color3.fromRGB(80,80,100)
        print(text .. ":", enabled)
    end)
end

createToggle("Freeze Trade", 40)
createToggle("Force Accept", 90)
