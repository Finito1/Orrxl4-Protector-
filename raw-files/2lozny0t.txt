local function MakeDraggable(frame, handle)
    local dragging, dragInput, dragStart, startPos
    handle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = frame.Position
        end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
            local delta = input.Position - dragStart
            frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
        end
    end)
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = false
        end
    end)
end

function OnyxUI:CreateToggle(parent, text, isLocked, callback)
    local frame = Instance.new("TextButton", parent)
    frame.Size = UDim2.new(1, -10, 0, 35)
    frame.BackgroundColor3 = Color3.fromRGB(20, 20, 25)
    frame.Text = "  " .. text
    frame.TextColor3 = Color3.fromRGB(255, 255, 255)
    frame.TextXAlignment = Enum.TextXAlignment.Left
    Instance.new("UICorner", frame)
    
    local state = false
    frame.MouseButton1Click:Connect(function()
        if isLocked then
            ShowPreviewAlert()
        else
            state = not state
            frame.BackgroundColor3 = state and Color3.fromRGB(170, 0, 255) or Color3.fromRGB(20, 20, 25)
            callback(state)
        end
    end)
end

-- [[ CORE LOGIC ]] --

RunService.Stepped:Connect(function()
    if lp.Character and lp.Character:FindFirstChild("HumanoidRootPart") then
        local hrp = lp.Character.HumanoidRootPart
        -- Logic is present but toggles for these are locked to false in preview
        if OnyxConfig.FakeLag then
            hrp.Anchored = true
        else
            hrp.Anchored = false
        end
    end
end)

RunService.RenderStepped:Connect(function()
    if not lp.Character or not lp.Character:FindFirstChild("HumanoidRootPart") then return end
    local hrp = lp.Character.HumanoidRootPart
    
    -- Fling logic (Inactive in Preview)
    if OnyxConfig.FlingOnTouch then
        local targetNear = false
        for _, p in pairs(Players:GetPlayers()) do
            if p ~= lp and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
                if (hrp.Position - p.Character.HumanoidRootPart.Position).Magnitude < 4.5 then
                    targetNear = true
                    break
                end
            end
        end
        if targetNear then
            hrp.CustomPhysicalProperties = PhysicalProperties.new(math.huge, 0.3, 0.5, 1, 1)
            hrp.AssemblyAngularVelocity = Vector3.new(0, 75000, 0)
            hrp.AssemblyLinearVelocity = Vector3.new(0, -100, 0)
        else
            hrp.CustomPhysicalProperties = PhysicalProperties.new(0.7, 0.3, 0.5, 1, 1)
        end
    end
end)

-- Infinite Jump (FUNCTIONAL)
UserInputService.JumpRequest:Connect(function()
    if OnyxConfig.InfJump and lp.Character and lp.Character:FindFirstChild("HumanoidRootPart") then
        lp.Character.HumanoidRootPart.CFrame = lp.Character.HumanoidRootPart.CFrame * CFrame.new(0, 2, 0)
    end
end)
