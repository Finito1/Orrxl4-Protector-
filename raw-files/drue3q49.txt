-- Click anim
local function clickAnim(btn)
    local orig = btn.BackgroundColor3
    TweenService:Create(btn, TweenInfo.new(0.08), {BackgroundColor3 = Color3.fromRGB(100, 100, 100)}):Play()
    task.wait(0.08)
    TweenService:Create(btn, TweenInfo.new(0.08), {BackgroundColor3 = orig}):Play()
end

-- Lógica Grab
local function getHRP()
    local char = player.Character
    return char and (char:FindFirstChild("HumanoidRootPart") or char:FindFirstChild("UpperTorso"))
end

local function scanPlot(plot)
    if not plot or not plot:IsA("Model") then return end
    local podiums = plot:FindFirstChild("AnimalPodiums")
    if not podiums then return end
    for _, pod in podiums:GetChildren() do
        if pod:IsA("Model") and pod:FindFirstChild("Base") then
            table.insert(allAnimalsCache, {
                plot = plot.Name,
                slot = pod.Name,
                worldPosition = pod:GetPivot().Position,
                uid = plot.Name .. "_" .. pod.Name,
            })
        end
    end
end

local function startScanner()
    task.wait(3)
    local plots = workspace:FindFirstChild("Plots")
    if plots then
        for _, p in plots:GetChildren() do scanPlot(p) end
        plots.ChildAdded:Connect(scanPlot)
    end
end

local function getPrompt(animal)
    local cache = PromptMemoryCache[animal.uid]
    if cache and cache.Parent then return cache end

    local plot = workspace.Plots:FindFirstChild(animal.plot)
    if not plot then return end
    local podium = plot.AnimalPodiums:FindFirstChild(animal.slot)
    if not podium then return end
    local base = podium:FindFirstChild("Base")
    if not base then return end
    local sp = base:FindFirstChild("Spawn")
    if not sp then return end
    local att = sp:FindFirstChild("PromptAttachment")
    if not att then return end
    local prompt = att:FindFirstChildOfClass("ProximityPrompt")
    if prompt then PromptMemoryCache[animal.uid] = prompt end
    return prompt
end

local function cacheCallbacks(prompt)
    if InternalStealCache[prompt] then return end
    local data = {hold={}, trigger={}}
    local s1, c1 = pcall(getconnections, prompt.PromptButtonHoldBegan)
    if s1 then for _, conn in c1 do table.insert(data.hold, conn.Function) end end
    local s2, c2 = pcall(getconnections, prompt.Triggered)
    if s2 then for _, conn in c2 do table.insert(data.trigger, conn.Function) end end
    InternalStealCache[prompt] = data
end

local function doGrab(prompt)
    if (os.clock() - LastGrabTime) < COOLDOWN_TIME then return end
    local data = InternalStealCache[prompt]
    if not data or IsStealing then return end

    IsStealing = true
    LastGrabTime = os.clock()

    task.spawn(function()
        for _, fn in data.hold do task.spawn(fn) end
        for _, fn in data.trigger do task.spawn(fn) end
        task.wait(COOLDOWN_TIME)
        IsStealing = false
    end)
end

local function nearestAnimal()
    local hrp = getHRP()
    if not hrp then return nil end
    local closest, minDist = nil, math.huge
    for _, animal in allAnimalsCache do
        local dist = (hrp.Position - animal.worldPosition).Magnitude
        if dist < minDist and dist <= GRAB_RADIUS then
            minDist = dist
            closest = animal
        end
    end
    return closest
end

instantBtn.MouseButton1Click:Connect(function()
    clickAnim(instantBtn)
    if IsStealing then return end
    local animal = nearestAnimal()
    if animal then
        local prompt = getPrompt(animal)
        if prompt then
            cacheCallbacks(prompt)
            doGrab(prompt)
        end
    end
end)

autoBtn.MouseButton1Click:Connect(function()
    AutoGrabEnabled = not AutoGrabEnabled
    clickAnim(autoBtn)
    autoBtn.Text = AutoGrabEnabled and "AUTO GRAB: ON" or "AUTO GRAB: OFF"
    autoBtn.BackgroundColor3 = AutoGrabEnabled and Color3.fromRGB(0, 170, 0) or Color3.fromRGB(50, 50, 50)
end)

task.spawn(function()
    while true do
        task.wait(0.06)
        if AutoGrabEnabled and not IsStealing and (os.clock() - LastGrabTime >= COOLDOWN_TIME) then
            local animal = nearestAnimal()
            if animal then
                task.wait(0.1)
                local check = nearestAnimal()
                if check and check.uid == animal.uid then
                    local p = getPrompt(check)
                    if p then
                        cacheCallbacks(p)
                        doGrab(p)
                    end
                end
            end
        end
    end
end)
