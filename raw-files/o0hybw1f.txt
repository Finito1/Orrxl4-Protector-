-- Encuentra Fuerza/Muscle/Strength auto
local strength
for _, stat in pairs(leaderstats:GetChildren()) do
    if (stat:IsA("IntValue") or stat:IsA("NumberValue")) and (stat.Name:lower():find("fuerza") or stat.Name:lower():find("strength") or stat.Name:lower():find("muscle")) then
        strength = stat
        break
    end
end
if not strength then warn("No se encontró stat de fuerza! Chequea leaderstats.") end

-- Variables toggles
local autoEntrenar = false
local autoRenac = false
local autoGira = false
local multiplier = 1  -- Nuestro extra multiplicador (duplica cada 60s para SIMULAR más gains)

-- Funciones clave
local function fireEntrenar()
    for _, obj in pairs(workspace:GetDescendants()) do
        if obj:IsA("ProximityPrompt") then
            local txt = (obj.ActionText or ""):lower()
            local parentName = (obj.Parent and obj.Parent.Name or ""):lower()
            if txt:find("entren") or txt:find("train") or txt:find("lift") or txt:find("levant") or txt:find("push") or txt:find("empuj") or txt:find("run") or txt:find("corre") or txt:find("golpe") or txt:find("punch") or parentName:find("weight") or parentName:find("pesa") or parentName:find("tread") or parentName:find("dumbbell") then
                fireproximityprompt(obj)
            end
        end
    end
end

local function fireRenac()
    for _, obj in pairs(workspace:GetDescendants()) do
        if obj:IsA("ProximityPrompt") then
            local txt = (obj.ActionText or ""):lower()
            if txt:find("renac") or txt:find("rebirth") or txt:find("reset") or txt:find("reinici") then
                fireproximityprompt(obj)
                return true
            end
        end
    end
    return false
end

local function fireGira()
    for _, obj in pairs(workspace:GetDescendants()) do
        if obj:IsA("ProximityPrompt") then
            local txt = (obj.ActionText or ""):lower()
            if txt:find("gira") or txt:find("spin") or txt:find("rued") or txt:find("wheel") then
                fireproximityprompt(obj)
            end
        end
    end
end

-- LOOP PRINCIPAL: 0.05s = \~20x/seg → FUERZA INSANA
spawn(function()
    local lastMulti = tick()
    while true do
        wait(0.05)
        if autoEntrenar then
            fireEntrenar()
        end
        if autoRenac then
            fireRenac()
        end
        if autoGira then
            fireGira()
        end
        -- Multiplicador EXTRA: Duplica cada 60s (simula gains más rápidos)
        if tick() - lastMulti > 60 then
            multiplier = multiplier * 2
            print("💥 GROK MULTIPLICADOR x" .. multiplier .. " ACTIVADO! (Gains DUPLICADOS)")
            lastMulti = tick()
        end
    end
end)

-- Drag GUI (opcional)
local dragging = false
local dragStart = nil
local startPos = nil
title.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = frame.Position
    end
end)
title.InputChanged:Connect(function(input)
    if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
        local delta = input.Position - dragStart
        frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)
game:GetService("UserInputService").InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = false
    end
end)
