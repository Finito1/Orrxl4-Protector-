--================ FUNCIÓN BASE PARA BOTONES =================
local function createButton(text, toggleKey)
	local btn = Instance.new("TextButton", Scroll)
	btn.Size = UDim2.new(1, -20, 0, 40)
	btn.Text = text
	btn.Font = Enum.Font.GothamBold
	btn.TextSize = 14
	btn.BackgroundColor3 = Color3.fromRGB(40,40,40)
	btn.TextColor3 = Color3.new(1,1,1)
	
	local corner = Instance.new("UICorner", btn)
	corner.CornerRadius = UDim.new(0,6)
	
	btn.MouseButton1Click:Connect(function()
		toggles[toggleKey] = not toggles[toggleKey]
		
		if toggles[toggleKey] then
			btn.BackgroundColor3 = Color3.fromRGB(220, 30, 30)  -- ON
		else
			btn.BackgroundColor3 = Color3.fromRGB(40,40,40)     -- OFF
		end
		
		if toggleKey == "AutoSteal" then toggleAutoSteal(toggles.AutoSteal) end
		if toggleKey == "ESPEnemigos" then toggleESPEnemigos(toggles.ESPEnemigos) end
		if toggleKey == "ESPBasesXray" then toggleESPBases(toggles.ESPBasesXray) end
		if toggleKey == "SpeedBoost" then toggleSpeed(toggles.SpeedBoost) end
		if toggleKey == "AntiRagdoll" then toggleAntiRagdoll(toggles.AntiRagdoll) end
		if toggleKey == "SuperSalto" then toggleSuperSalto(toggles.SuperSalto) end
	end)
	
	-- Color inicial según estado (por si reload)
	if toggles[toggleKey] then
		btn.BackgroundColor3 = Color3.fromRGB(220, 30, 30)
	else
		btn.BackgroundColor3 = Color3.fromRGB(40,40,40)
	end
	
	return btn
end

-- AUTO GRAB (sin cambios)
local autoStealConn
function toggleAutoSteal(enable)
    if autoStealConn then autoStealConn:Disconnect() autoStealConn = nil end
    if not enable then return end
    
    autoStealConn = RunService.Heartbeat:Connect(function()
        local char = LocalPlayer.Character
        if not char or not char:FindFirstChild("HumanoidRootPart") then return end
        local hrp = char.HumanoidRootPart
        
        local myPlotName = "Plot_" .. LocalPlayer.Name
        for _, plot in ipairs(Workspace:FindFirstChild("Plots") and Workspace.Plots:GetChildren() or {}) do
            if plot.Name == myPlotName then continue end
            
            if (plot:GetPivot().Position - hrp.Position).Magnitude > 80 then continue end
            
            local podiums = plot:FindFirstChild("AnimalPodiums")
            if not podiums then continue end
            
            for _, podium in ipairs(podiums:GetChildren()) do
                local basePart = podium:FindFirstChild("Base")
                if not basePart then continue end
                
                local spawnPart = basePart:FindFirstChild("Spawn")
                if not spawnPart then continue end
                
                local promptAttach = spawnPart:FindFirstChild("PromptAttachment")
                local prompt = promptAttach and promptAttach:FindFirstChildOfClass("ProximityPrompt")
                
                if prompt and (spawnPart.Position - hrp.Position).Magnitude < 15 then
                    prompt.HoldDuration = 0
                    fireproximityprompt(prompt)
                    task.wait(0.15)
                end
            end
        end
    end)
end

-- ESP ENEMIGOS (sin cambios)
local enemyESPs = {}
local espPlayerAddedConn

function toggleESPEnemigos(enable)
    if not enable then
        for _, esp in ipairs(enemyESPs) do pcall(function() esp:Destroy() end) end
        enemyESPs = {}
        if espPlayerAddedConn then espPlayerAddedConn:Disconnect() espPlayerAddedConn = nil end
        return
    end
    
    for _, esp in ipairs(enemyESPs) do pcall(function() esp:Destroy() end) end
    enemyESPs = {}
    
    local function addESP(plr)
        if plr == LocalPlayer or not plr.Character then return end
        local char = plr.Character
        local head = char:FindFirstChild("Head")
        if not head or head:FindFirstChild("NameESP") then return end
        
        local hl = Instance.new("Highlight")
        hl.FillColor = Color3.fromRGB(255, 80, 80)
        hl.OutlineColor = Color3.fromRGB(255, 0, 0)
        hl.FillTransparency = 0.5
        hl.OutlineTransparency = 0
        hl.Adornee = char
        hl.Parent = char
        table.insert(enemyESPs, hl)
        
        local bb = Instance.new("BillboardGui")
        bb.Name = "NameESP"
        bb.Adornee = head
        bb.Size = UDim2.new(0, 120, 0, 30)
        bb.StudsOffset = Vector3.new(0, 3.2, 0)
        bb.AlwaysOnTop = true
        bb.Parent = head
        
        local text = Instance.new("TextLabel", bb)
        text.Size = UDim2.new(1,0,1,0)
        text.BackgroundTransparency = 1
        text.Text = plr.Name
        text.TextColor3 = Color3.new(1,1,1)
        text.TextStrokeTransparency = 0
        text.TextSize = 14
        text.Font = Enum.Font.GothamBold
        
        table.insert(enemyESPs, bb)
    end
    
    for _, plr in ipairs(Players:GetPlayers()) do addESP(plr) end
    
    espPlayerAddedConn = Players.PlayerAdded:Connect(addESP)
end

-- ESP BASES X-RAY (sin cambios)
local baseESPs = {}
local originalTransparencies = {}

function toggleESPBases(enable)
    local plots = Workspace:FindFirstChild("Plots")
    if not plots then return end
    
    if not enable then
        for _, esp in ipairs(baseESPs) do pcall(function() esp:Destroy() end) end
        baseESPs = {}
        
        for plotName, data in pairs(originalTransparencies) do
            local plot = plots:FindFirstChild(plotName)
            if plot then
                for part, originalTrans in pairs(data) do
                    pcall(function() part.Transparency = originalTrans end)
                end
            end
        end
        originalTransparencies = {}
        return
    end
    
    originalTransparencies = {}
    baseESPs = {}
    
    for _, plot in ipairs(plots:GetChildren()) do
        originalTransparencies[plot.Name] = {}
        local transTable = originalTransparencies[plot.Name]
        
        for _, part in ipairs(plot:GetDescendants()) do
            if part:IsA("BasePart") or part:IsA("MeshPart") then
                if part.Transparency < 0.6 then
                    transTable[part] = part.Transparency
                    part.Transparency = 0.68
                end
            end
        end
        
        local hl = Instance.new("Highlight")
        hl.Name = "BaseXRayHighlight"
        hl.FillColor = Color3.fromRGB(0, 240, 210)
        hl.OutlineColor = Color3.fromRGB(0, 180, 255)
        hl.FillTransparency = 0.78
        hl.OutlineTransparency = 0.15
        hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
        hl.Adornee = plot
        hl.Parent = plot
        table.insert(baseESPs, hl)
    end
end

-- SPEED BOOST (sin cambios)
local speedConn
function toggleSpeed(enable)
    if speedConn then 
        speedConn:Disconnect() 
        speedConn = nil 
    end
    
    local char = LocalPlayer.Character
    local hum = char and char:FindFirstChildWhichIsA("Humanoid")
    
    if hum then
        hum.WalkSpeed = 16  -- Siempre reset al cambiar
    end
    
    if not enable then return end
    
    speedConn = RunService.Heartbeat:Connect(function()
        if not toggles.SpeedBoost then
            if speedConn then speedConn:Disconnect() speedConn = nil end
            return
        end
        
        local char = LocalPlayer.Character
        if not char then return end
        
        local hum = char:FindFirstChildWhichIsA("Humanoid")
        local hrp = char:FindFirstChild("HumanoidRootPart")
        if not hum or not hrp then return end
        
        if hum.PlatformStand then
            hum.PlatformStand = false
        end
        
        hum.WalkSpeed = 28
        
        if hum.MoveDirection.Magnitude > 0 then
            local currentVel = hrp.AssemblyLinearVelocity
            local targetVel = hum.MoveDirection * hum.WalkSpeed
            hrp.AssemblyLinearVelocity = Vector3.new(targetVel.X, currentVel.Y, targetVel.Z)
        end
    end)
end

-- NUEVA VERSIÓN: INFINITE JUMP (reemplaza totalmente el super salto anterior)
-- Funciona presionando espacio varias veces seguidas → salta en el aire
local infJumpConn
function toggleSuperSalto(enable)
    if infJumpConn then 
        infJumpConn:Disconnect() 
        infJumpConn = nil 
    end
    
    if not enable then return end
    
    infJumpConn = UserInputService.JumpRequest:Connect(function()
        if not toggles.SuperSalto then return end
        
        local character = LocalPlayer.Character
        if not character then return end
        
        local humanoid = character:FindFirstChildOfClass("Humanoid")
        local hrp = character:FindFirstChild("HumanoidRootPart")
        
        if not humanoid or not hrp then return end
        
        -- Evitamos estados que rompen el salto
        local state = humanoid:GetState()
        if state == Enum.HumanoidStateType.Dead then return end
        
        if humanoid.PlatformStand then 
            humanoid.PlatformStand = false 
        end
        
        -- Aplicamos impulso hacia arriba (funciona en aire y en suelo)
        local currentVel = hrp.AssemblyLinearVelocity
        hrp.AssemblyLinearVelocity = Vector3.new(currentVel.X, 50, currentVel.Z)  -- 50 es un buen valor alto pero no extremo
    end)
end

-- ANTI RAGDOLL (sin cambios)
local antiRagdollConn
function toggleAntiRagdoll(enable)
    if antiRagdollConn then 
        antiRagdollConn:Disconnect() 
        antiRagdollConn = nil 
    end
    
    if not enable then 
        local char = LocalPlayer.Character
        if char then
            local hum = char:FindFirstChildWhichIsA("Humanoid")
            if hum then
                hum:SetStateEnabled(Enum.HumanoidStateType.Ragdoll, true)
                hum:SetStateEnabled(Enum.HumanoidStateType.FallingDown, true)
                hum:SetStateEnabled(Enum.HumanoidStateType.Physics, true)
            end
        end
        return 
    end
    
    local char = LocalPlayer.Character
    local hum = char and char:FindFirstChildWhichIsA("Humanoid")
    
    if hum then
        hum:SetStateEnabled(Enum.HumanoidStateType.Ragdoll, false)
        hum:SetStateEnabled(Enum.HumanoidStateType.FallingDown, false)
        hum:SetStateEnabled(Enum.HumanoidStateType.Physics, false)
    end
    
    antiRagdollConn = RunService.Heartbeat:Connect(function()
        local char = LocalPlayer.Character
        if not char then return end
        
        local hum = char:FindFirstChildWhichIsA("Humanoid")
        if not hum then return end
        
        if hum.PlatformStand then hum.PlatformStand = false end
        
        local state = hum:GetState()
        
        if state == Enum.HumanoidStateType.Ragdoll or state == Enum.HumanoidStateType.Physics or state == Enum.HumanoidStateType.FallingDown or state == Enum.HumanoidStateType.Dead then
            hum:ChangeState(Enum.HumanoidStateType.GettingUp)
            
            task.spawn(function()
                task.wait(0.01)
                if hum and hum.Parent then
                    hum:ChangeState(Enum.HumanoidStateType.Running)
                end
            end)
            
            task.delay(0.03, function()
                if hum and hum.Parent and hum:GetState() ~= Enum.HumanoidStateType.Running then
                    hum:ChangeState(Enum.HumanoidStateType.Running)
                end
            end)
        end
    end)
end

    local hum = char:WaitForChild("Humanoid", 5)

    -- Solo activamos si el toggle está true

    if toggles.SuperSalto then toggleSuperSalto(true) end

    if toggles.ESPEnemigos then 
        toggleESPEnemigos(false)
        task.wait(0.1)
        toggleESPEnemigos(true)
    end
    
    if toggles.ESPBasesXray then 
        toggleESPBases(false)
        task.wait(0.1)
        toggleESPBases(true)
    end
end)
